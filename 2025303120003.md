## 3.1 元数据定义（数据字典+编码规范）
define_metadata <- function() {
  metadata <- list(
    # 核心数据字段说明
    data_dictionary = data.frame(
      field = c("sample_id", "site", "land_use", "depth_cm", "polymer", 
                "shape", "color", "size_um", "count", "soil_dry_weight_g"),
      description = c("Unique sample identifier",
                      "Sampling site name",
                      "Land use type (Agricultural/Forest/Urban/Industrial)",
                      "Sampling depth in cm",
                      "Polymer type (PE/PP/PS/PET/PVC)",
                      "Particle shape (Fiber/Fragment/Film/Foam/Pellet)",
                      "Particle color",
                      "Particle size in micrometers",
                      "Number of particles counted",
                      "Dry weight of soil sample in grams"),
      unit = c("-", "-", "-", "cm", "-", "-", "-", "μm", "items", "g"),
      stringsAsFactors = FALSE
    ),
    # 聚合物编码规范
    polymer_codes = c(
      "PE" = "Polyethylene", 
      "PP" = "Polypropylene", 
      "PS" = "Polystyrene", 
      "PET" = "Polyethylene terephthalate",
      "PVC" = "Polyvinyl chloride", 
      "PA" = "Polyamide"
    ),
    # 形状编码规范
    shape_codes = c("Fiber", "Fragment", "Film", "Foam", "Pellet"),
    # 颜色编码规范
    color_codes = c("Transparent", "White", "Black", "Blue", "Red", "Green", "Yellow", "Mixed")
  )
  return(metadata)
}

## 3.2 数据导入（兼容缺失文件）
import_data <- function() {
  cat("Importing raw data...\n")
  
  # 读取采样元数据
  sample_info <- tryCatch(
    read.csv(here::here("data/raw/sampling_metadata.csv"), stringsAsFactors = FALSE),
    error = function(e) {
      cat("Warning: sampling_metadata.csv not found (data/raw/). Using empty template.\n")
      data.frame(sample_id = character(), stringsAsFactors = FALSE)
    }
  )
  
  # 读取微塑料计数数据
  mp_counts <- tryCatch(
    read.csv(here::here("data/raw/mp_counts.csv"), stringsAsFactors = FALSE),
    error = function(e) {
      cat("Warning: mp_counts.csv not found (data/raw/). Using empty template.\n")
      data.frame(sample_id = character(), stringsAsFactors = FALSE)
    }
  )
  
  # 读取空白对照数据
  blanks <- tryCatch(
    read.csv(here::here("data/raw/blanks.csv"), stringsAsFactors = FALSE),
    error = function(e) {
      cat("Warning: blanks.csv not found (data/raw/). Using default blank value (count=0).\n")
      data.frame(sample_id = "blank", count = 0, stringsAsFactors = FALSE)
    }
  )
  
  # 返回数据列表
  return(list(samples = sample_info, mp = mp_counts, blanks = blanks))
}

## 3.3 数据清洗与标准化
clean_data <- function(data_list) {
  cat("Cleaning and standardizing data...\n")
  
  # 加载元数据规范
  metadata <- define_metadata()
  
  # 数据合并+标准化
  cleaned_data <- data_list$mp %>%
    # 合并采样信息（按sample_id关联）
    dplyr::left_join(data_list$samples, by = "sample_id") %>%
    # 计算每千克干土的丰度（核心指标）
    dplyr::mutate(
      abundance_kg_dw = count / (soil_dry_weight_g / 1000),
      .after = count
    ) %>%
    # 标准化聚合物名称（兼容大小写/全称缩写）
    dplyr::mutate(
      polymer_std = dplyr::case_when(
        stringr::str_detect(tolower(polymer), "polyethylene|pe") ~ "PE",
        stringr::str_detect(tolower(polymer), "polypropylene|pp") ~ "PP",
        stringr::str_detect(tolower(polymer), "polystyrene|ps") ~ "PS",
        stringr::str_detect(tolower(polymer), "pet|terephthalate") ~ "PET",
        stringr::str_detect(tolower(polymer), "pvc|polyvinyl") ~ "PVC",
        TRUE ~ "Others"
      ),
      polymer_std = factor(polymer_std, levels = c("PE", "PP", "PS", "PET", "PVC", "Others")),
      .after = polymer
    ) %>%
    # 标准化形状名称
    dplyr::mutate(
      shape_std = factor(tolower(shape), levels = tolower(metadata$shape_codes)),
      .after = shape
    ) %>%
    # 标准化土地利用类型
    dplyr::mutate(
      land_use = factor(land_use, levels = c("Agricultural", "Forest", "Urban", "Industrial"))
    ) %>%
    # 移除完全缺失的行
    tidyr::drop_na(sample_id)
  
  return(cleaned_data)
}

## 4.1 空白对照扣除（计算LOD并修正丰度）
apply_blank_correction <- function(data, blanks) {
  cat("Applying blank correction (LOD calculation)...\n")
  
  # 计算空白对照统计量（均值+标准差+检测限LOD）
  blank_stats <- blanks %>%
    dplyr::summarise(
      mean_blank = mean(count, na.rm = TRUE),
      sd_blank = sd(count, na.rm = TRUE),
      lod = mean_blank + 3 * sd_blank,  # 检测限=均值+3倍标准差
      n_blank = dplyr::n(),
      .groups = "drop"
    )
  
  # 打印空白统计结果
  cat(sprintf("Blank control stats: Mean=%.2f, SD=%.2f, LOD=%.2f (n=%d)\n", 
              blank_stats$mean_blank, blank_stats$sd_blank, 
              blank_stats$lod, blank_stats$n_blank))
  
  # 应用空白扣除（低于LOD设为0，避免负值）
  data_corrected <- data %>%
    dplyr::mutate(
      abundance_net = dplyr::if_else(
        abundance_kg_dw <= blank_stats$lod,
        0,
        abundance_kg_dw - blank_stats$mean_blank
      ),
      abundance_net = pmax(abundance_net, 0),  # 确保非负
      qc_flag = dplyr::case_when(
        abundance_kg_dw <= blank_stats$lod ~ "Below LOD",
        TRUE ~ "Above LOD"
      ),
      .after = abundance_kg_dw
    )
  
  # 保存空白统计结果
  readr::write_csv(blank_stats, here::here("outputs/tables/blank_statistics.csv"))
  
  return(list(data = data_corrected, stats = blank_stats))
}

## 4.2 数据质量验证（缺失值/负值/异常值检查）
validate_data_quality <- function(data) {
  cat("Validating data quality...\n")
  
  quality_issues <- list()
  
  # 1. 检查缺失值
  missing_values <- colSums(is.na(data))
  if (any(missing_values > 0)) {
    quality_issues$missing_values <- missing_values[missing_values > 0]
  }
  
  # 2. 检查负值（丰度不应为负）
  negative_abundance <- sum(data$abundance_net < 0, na.rm = TRUE)
  if (negative_abundance > 0) {
    quality_issues$negative_abundance <- negative_abundance
  }
  
  # 3. 检查极端异常值（IQR*3准则）
  q75 <- quantile(data$abundance_net, 0.75, na.rm = TRUE)
  q25 <- quantile(data$abundance_net, 0.25, na.rm = TRUE)
  iqr <- q75 - q25
  outliers <- sum(data$abundance_net > q75 + 3 * iqr, na.rm = TRUE)
  if (outliers > 0) {
    quality_issues$outliers <- outliers
  }
  
  # 输出检查结果
  if (length(quality_issues) == 0) {
    cat("✅ Data quality check passed (no issues found).\n")
  } else {
    cat("⚠️ Data quality issues found:\n")
    print(quality_issues)
  }
  
  return(quality_issues)
}

## 5.1 描述性统计（按土地利用/深度分组）
calculate_descriptive_stats <- function(data) {
  cat("Calculating descriptive statistics...\n")
  
  desc_stats <- data %>%
    dplyr::group_by(land_use, depth_cm) %>%
    dplyr::summarise(
      n_samples = dplyr::n(),
      n_detected = sum(abundance_net > 0, na.rm = TRUE),
      detection_rate = round(n_detected / n_samples * 100, 2),
      total_abundance = sum(abundance_net, na.rm = TRUE),
      mean_abundance = mean(abundance_net, na.rm = TRUE),
      median_abundance = median(abundance_net, na.rm = TRUE),
      sd_abundance = sd(abundance_net, na.rm = TRUE),
      se_abundance = sd_abundance / sqrt(n_samples),
      iqr_abundance = IQR(abundance_net, na.rm = TRUE),
      min_abundance = min(abundance_net, na.rm = TRUE),
      max_abundance = max(abundance_net, na.rm = TRUE),
      .groups = "drop"
    )
  
  return(desc_stats)
}

## 5.2 差异性检验（非参数检验，适配生态学数据）
run_statistical_tests <- function(data) {
  cat("Running statistical tests (non-parametric)...\n")
  
  test_results <- list()
  
  # 过滤有效数据（避免空值）
  valid_data <- data %>% dplyr::filter(!is.na(land_use) & abundance_net >= 0)
  
  # 多组比较：Kruskal-Wallis检验（替代ANOVA，非参数）
  if (nlevels(valid_data$land_use) > 2) {
    kw_test <- kruskal.test(abundance_net ~ land_use, data = valid_data)
    test_results$kruskal_wallis <- broom::tidy(kw_test)
    
    # 两两比较（Mann-Whitney U + Bonferroni校正）
    land_use_groups <- levels(valid_data$land_use)
    pairwise_comparisons <- utils::combn(land_use_groups, 2, function(pair) {
      subset_data <- valid_data %>% dplyr::filter(land_use %in% pair)
      mw_test <- wilcox.test(abundance_net ~ land_use, data = subset_data)
      data.frame(
        group1 = pair[1],
        group2 = pair[2],
        p_value = mw_test$p.value,
        p_adjusted = p.adjust(mw_test$p.value, method = "bonferroni"),
        stringsAsFactors = FALSE
      )
    })
    test_results$pairwise_comparisons <- do.call(rbind, pairwise_comparisons)
    
    # 两组比较：Mann-Whitney U检验
  } else if (nlevels(valid_data$land_use) == 2) {
    mw_test <- wilcox.test(abundance_net ~ land_use, data = valid_data)
    test_results$mann_whitney <- broom::tidy(mw_test)
  } else {
    cat("⚠️ Insufficient groups for statistical tests (need ≥2 land use types).\n")
  }
  
  return(test_results)
}

## 5.3 多元分析（NMDS+PERMANOVA，聚合物群落结构）
run_multivariate_analysis <- function(data) {
  cat("Running multivariate analysis (NMDS + PERMANOVA)...\n")
  
  # 构建聚合物群落矩阵（sample × polymer）
  comm_matrix <- data %>%
    dplyr::select(sample_id, polymer_std, abundance_net) %>%
    tidyr::pivot_wider(
      names_from = polymer_std,
      values_from = abundance_net,
      values_fill = 0,
      id_cols = sample_id
    ) %>%
    tibble::column_to_rownames("sample_id") %>%
    as.matrix()
  
  # 过滤空样本（无微塑料的样本）
  comm_matrix <- comm_matrix[rowSums(comm_matrix) > 0, , drop = FALSE]
  
  # 若矩阵为空，跳过分析
  if (nrow(comm_matrix) == 0) {
    cat("⚠️ No valid samples for multivariate analysis (all abundance = 0).\n")
    return(NULL)
  }
  
  # NMDS分析（Bray-Curtis距离，2维）
  set.seed(42)  # 固定随机种子，确保结果可重复
  nmds_result <- vegan::metaMDS(comm_matrix, distance = "bray", k = 2, trymax = 200)
  
  # PERMANOVA（检验土地利用对群落结构的影响）
  meta_data <- data %>%
    dplyr::filter(sample_id %in% rownames(comm_matrix)) %>%
    dplyr::select(sample_id, land_use, depth_cm) %>%
    tibble::column_to_rownames("sample_id")
  meta_data <- meta_data[match(rownames(comm_matrix), rownames(meta_data)), ]
  
  permanova_result <- vegan::adonis2(
    comm_matrix ~ land_use,
    data = meta_data,
    permutations = 999,
    method = "bray"
  )
  
  # 提取NMDS坐标（用于绘图）
  nmds_scores <- as.data.frame(vegan::scores(nmds_result, display = "sites")) %>%
    tibble::rownames_to_column("sample_id") %>%
    dplyr::left_join(meta_data %>% tibble::rownames_to_column("sample_id"), by = "sample_id")
  
  # 返回分析结果
  return(list(
    nmds = nmds_result,
    nmds_scores = nmds_scores,
    permanova = permanova_result,
    community_matrix = comm_matrix
  ))
}

## 6.2 聚合物组成堆叠图（按样本）
plot_polymer_composition <- function(data) {
  # 计算每个样本的聚合物相对丰度
  composition_data <- data %>%
    dplyr::group_by(sample_id, polymer_std) %>%
    dplyr::summarise(total_count = sum(count, na.rm = TRUE), .groups = "drop") %>%
    dplyr::group_by(sample_id) %>%
    dplyr::mutate(relative_abundance = total_count / sum(total_count)) %>%
    dplyr::ungroup()
  
  p <- ggplot2::ggplot(composition_data, ggplot2::aes(x = sample_id, y = relative_abundance, fill = polymer_std)) +
    ggplot2::geom_bar(position = "fill", stat = "identity", width = 0.8) +
    ggplot2::scale_fill_brewer(palette = "Set3", name = "Polymer Type") +
    ggplot2::labs(
      title = "Relative Abundance of Polymer Types by Sample",
      y = "Relative Abundance",
      x = "Sample ID"
    ) +
    ggplot2::theme_minimal() +
    ggplot2::theme(
      plot.title = ggplot2::element_text(hjust = 0.5, face = "bold", size = 12),
      legend.position = "bottom",
      axis.text.x = ggplot2::element_text(angle = 90, hjust = 1, size = 7)
    )
  
  return(p)
}

# ============================================
blank_correction <- function(data, blanks) {
  # 计算检测限 (LOD)
  blank_stats <- blanks %>%
    summarise(
      mean_blank = mean(count, na.rm = TRUE),
      sd_blank = sd(count, na.rm = TRUE),
      lod = mean_blank + 3 * sd_blank
    )
  
  # 空白扣除
  data_corrected <- data %>%
    mutate(
      abundance_corrected = ifelse(abundance <= blank_stats$lod, 
                                    0, 
                                    abundance - blank_stats$mean_blank),
      # 单位换算：items/kg 干重
      abundance_kg_dw = abundance_corrected / (soil_dry_weight_g / 1000)
    )
  
  return(data_corrected)
}

# NMDS 多元分析
nmds_analysis <- function(data) {

  comm_matrix <- data %>%
    select(sample_id, polymer, abundance_kg_dw) %>%
    pivot_wider(names_from = polymer, values_from = abundance_kg_dw, values_fill = 0) %>%
    column_to_rownames("sample_id")
  
  # NMDS
  nmds <- metaMDS(comm_matrix, distance = "bray", k = 2, trymax = 100)
  
  # 提取结果
  nmds_scores <- as.data.frame(scores(nmds, display = "sites")) %>%
    rownames_to_column("sample_id")
  
  return(list(nmds = nmds, scores = nmds_scores))
}

# 5.2 组间差异检验
run_group_tests <- function(data) {
  # Kruskal-Wallis
  kw_test <- kruskal.test(abundance_corrected ~ land_use, data = data)
  
  # 两两比较 (Wilcoxon with Bonferroni)
  pairs <- combn(levels(data$land_use), 2)
  pairwise_results <- apply(pairs, 2, function(p) {
    subset_data <- subset(data, land_use %in% p)
    test <- wilcox.test(abundance_corrected ~ land_use, data = subset_data)
    data.frame(
      group1 = p[1],
      group2 = p[2],
      p.value = test$p.value
    )
  })
  
  list(kruskal = kw_test, pairwise = pairwise_results)
}

# 5.3 NMDS 群落分析
run_nmds <- function(data) {
  # 构建物种-样地矩阵
  comm_matrix <- data %>%
    select(sample_id, polymer, abundance_corrected) %>%
    pivot_wider(names_from = polymer, values_from = abundance_corrected, values_fill = 0) %>%
    column_to_rownames("sample_id") %>%
    as.matrix()
  
  # NMDS
  set.seed(123)
  nmds <- metaMDS(comm_matrix, distance = "bray", k = 2, trymax = 200)
  
  # PERMANOVA
  perm <- adonis2(comm_matrix ~ land_use, data = data.frame(land_use = data$land_use[1:nrow(comm_matrix)]), 
                  permutations = 999)
  
  list(nmds = nmds, permanova = perm, matrix = comm_matrix)
}
